219 {
220  std::vector<ifloat> heap;
221  std::map<int, int> in_heap;
222  std::stack<int> stack;
223  float max_dist, dist;
224  int cmp = 0;
225 
226  ASSERT_FATAL_ERROR(query < _data->size(), "Data doesn't contains the query");
227 
228  for(int i=0; i < k; i++) {// Sets the heap with initial guesses
229  heap.push_back(ifloat(i*_dim, _metric(query, i*_dim, *_data, _dim)));
230  in_heap[i*_dim] = i*_dim;
231  }
232 
233  std::make_heap(heap.begin(), heap.end());
234  max_dist = heap.front().val();
235  stack.push(cmp);
236  while(!stack.empty())
237  {

239 
240  dist = _metric(query, _tree[cmp]._key, *_data, _dim);
241 
242  if(_tree[cmp]._lc == LEAF && _tree[cmp]._rc == LEAF) {// if leaf
243  if(dist < max_dist && !in_heap.count(_tree[cmp]._key)) {
244  in_heap.erase(heap.front().key());
245  std::pop_heap(heap.begin(), heap.end()); heap.pop_back();
246  heap.push_back(ifloat(_tree[cmp]._key, dist));
247  in_heap[_tree[cmp]._key] = _tree[cmp]._key;
248  std::push_heap(heap.begin(), heap.end());
249  max_dist = heap.front().val();
250  }
251  }
252  else // if node is not leaf
253  {
254  if (dist <= _tree[cmp]._d + max_dist)
255  stack.push(_tree[cmp]._lc);
256  if (dist >= _tree[cmp]._d - max_dist)
257  stack.push(_tree[cmp]._rc);
258  }
259  }

261  id.clear();
262  for(int i=0; i < heap.size(); i++)
263  id.push_back(heap[i].key());

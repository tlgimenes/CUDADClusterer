129 inline int VPTreeCPU::make_vp_tree(std::vector<ifloat>& index_set)
130 {
131  std::vector<ifloat> l_set, r_set;
132  int p = 0;
133  float mu = 0.0f;
134  int middle = index_set.size() / 2;
135 
136  if(index_set.size() == 1) {
137  _tree.push_back(vp_node(index_set[0].key(), 0, -1, -1));
138  return _tree.size()-1;
139  }
140 
141  p = this->select_vp(index_set);
142  this->dist2(p, index_set);
143  std::sort(index_set.begin(), index_set.end());
144  mu = index_set[middle].val();
145 
146  l_set.assign(index_set.begin(), index_set.begin()+middle);
147  r_set.assign(index_set.begin()+middle, index_set.end());
148 
149  _tree.push_back(vp_node(p, mu, make_vp_tree(l_set), make_vp_tree(r_set)));
150 
151  return _tree.size()-1;
152 }

189 inline float VPTreeCPU::dist2(int a, int b)
190 {
191  float res = 0.0f;
192 
193  ASSERT_FATAL_ERROR((a+_dim <= _data.size()) && (b+_dim <= _data.size()), "Out of bounds");
194 
195  for(int i=0; i < _dim; i++)
196  {
197  res += (_data[a+i] - _data[b+i]) * (_data[a+i] - _data[b+i]);
198  }
199 
200  return res;
201 }

91  ReaderXTC::get_framefile_list(trajlist, home, trajlist_path);

122 inline void ReaderXTC::get_framefile_list(std::vector<std::string>& trajlist,
123  const std::string& home, const std::string& trajlinks_path)
124 {
125  std::stringstream trajlinks;
126  std::string sub_traj;
127 
128  // Recursion base, when file is a complete path
129  if(ReaderXTC::is_ext_supported(trajlinks_path))
130  trajlist.push_back(home + trajlinks_path);
131  // when file is a link to a file
132  else {
133  // Gets contents of the file
134  trajlinks << std::ifstream(home + trajlinks_path, std::ios_base::in).rdbuf();
135  trajlinks >> sub_traj;
136 
137  while(trajlinks && !trajlinks.eof() && sub_traj.size() > 0) // Foreach link in the file
138  {
139  if(bfs::is_regular_file(home+sub_traj)) // if file is openable and is a link to other files
140  ReaderXTC::get_framefile_list(trajlist, home, sub_traj);
141  else if(bfs::is_regular_file(home+sub_traj+".xtc")) // if file without xtc extension
142  trajlist.push_back(home+sub_traj+".xtc");
143  else // default file name
144  trajlist.push_back(home+sub_traj+"/frame0.xtc");
145 
146  trajlinks >> sub_traj; // gets new file/link
147  }
148  }
149 }

153 inline void ReaderXTC::read_trajfile(const std::string& trajfile,
154  std::vector<float> &data, int& n_atoms, int& n_samples)
155 {
156  int step;
157  float time, prec;
158  matrix box;
159  rvec* tmp = new rvec[n_atoms];
160  n_samples = 0;
161 
162  // Opens trajfile
163  XDRFILE *xdr_file = xdrfile_open(trajfile.c_str(), "r"); // opens file
164 
165  // Reads each frame
166  while (exdrOK == read_xtc(xdr_file, n_atoms, &step, &time, box, tmp, &prec)) {
167  for(int i=0; i < n_atoms; i++) {
168  data.push_back(tmp[i][0]);
169  data.push_back(tmp[i][1]);
170  data.push_back(tmp[i][2]);
171  }
172  n_samples++;
173  }
174 
175  // Closes trajfile
176  xdrfile_close(xdr_file);
177 
178  // Deletes allocated memory
179  delete []tmp;
180 }
